/**
 * @fileoverview Email verification code management service.
 *
 * Handles the full lifecycle of one-time verification codes used to confirm
 * user email addresses during registration and email-change flows.
 *
 * **Business logic overview:**
 * 1. A 6-digit numeric code is generated and stored in the
 *    `EmailVerificationCodes` table with a 15-minute TTL.
 * 2. Any previously active code for the same email is invalidated before a
 *    new one is issued (prevents code stuffing).
 * 3. On successful verification the code is marked `is_used = true` and can
 *    no longer be used.
 *
 * @module services/verificationService
 */

import { query } from '../config/database.js';

/**
 * Service responsible for generating, storing and validating one-time
 * numeric verification codes.
 */
class VerificationService {
  /**
   * Generates a cryptographically-weak but sufficiently random 6-digit code.
   *
   * **Algorithm:** `Math.floor(100000 + Math.random() * 900000)` guarantees
   * the result is always in the range [100000, 999999] (6 digits, no leading
   * zero).  For higher security requirements consider `crypto.randomInt`.
   *
   * @returns {string} A 6-digit numeric string, e.g. `'483920'`.
   *
   * @example
   * const code = service.generateVerificationCode();
   * console.log(code); // '483920'
   */
  generateVerificationCode() {
    return Math.floor(100000 + Math.random() * 900000).toString();
  }

  /**
   * Creates a new verification code for the given email and persists it in
   * the database.
   *
   * Any existing un-used codes for `email` are invalidated first to ensure
   * only one active code exists at a time.
   *
   * @param {string} email              - The recipient email address.
   * @param {string} [verification_type='registration'] - The reason for
   *   verification; currently `'registration'` or `'email_change'`.
   * @returns {Promise<string>} Resolves with the plain-text code that was
   *   stored (so the caller can forward it via email).
   * @throws {Error} Throws `'Failed to create verification code'` if the
   *   database operation fails.
   *
   * @example
   * const code = await verificationService.createVerificationCode(
   *   'user@example.com',
   *   'registration'
   * );
   * await emailService.sendVerificationEmail('user@example.com', code);
   */
  async createVerificationCode(email, verification_type = 'registration') {
    const code = this.generateVerificationCode();
    const expiresAt = new Date(Date.now() + 15 * 60 * 1000); // 15 minutes from now

    try {
      // Invalidate any existing codes for this email
      await query(
        'UPDATE EmailVerificationCodes SET is_used = true WHERE email = $1 AND is_used = false',
        [email]
      );

      // Insert new code
      await query(
        'INSERT INTO EmailVerificationCodes (email, code, expires_at, verification_type) VALUES ($1, $2, $3, $4)',
        [email, code, expiresAt, verification_type]
      );

      return code;
    } catch (error) {
      console.error('Error creating verification code:', error);
      throw new Error('Failed to create verification code');
    }
  }

  /**
   * Validates a verification code submitted by the user.
   *
   * The code is valid only if it exists in the database, has not been used,
   * and has not yet expired.  On successful validation the code is marked as
   * used so it cannot be replayed.
   *
   * @param {string} email - The email address associated with the code.
   * @param {string} code  - The 6-digit code submitted by the user.
   * @returns {Promise<{isValid: boolean, message: string}>} An object
   *   describing the outcome.  `isValid` is `true` only on success.
   *
   * @example
   * const result = await verificationService.verifyCode('user@example.com', '483920');
   * if (!result.isValid) {
   *   return res.status(400).json({ error: result.message });
   * }
   */
  async verifyCode(email, code) {
    try {
      const result = await query(
        `SELECT * FROM EmailVerificationCodes
         WHERE email = $1
         AND code = $2
         AND is_used = false
         AND expires_at > NOW()`,
        [email, code]
      );

      if (result.rows.length === 0) {
        return { isValid: false, message: 'Невірний або прострочений код' };
      }

      // Mark code as used
      await query(
        'UPDATE EmailVerificationCodes SET is_used = true WHERE email = $1 AND code = $2',
        [email, code]
      );

      return { isValid: true, message: 'Код підтверджено успішно' };
    } catch (error) {
      console.error('Error verifying code:', error);
      return { isValid: false, message: 'Помилка перевірки коду' };
    }
  }

  /**
   * Checks whether there is an active (non-expired, non-used) verification
   * code for the given email.
   *
   * Used by the registration flow to prevent sending multiple codes in quick
   * succession, which would confuse users and waste email quota.
   *
   * @param {string} email - The email address to check.
   * @returns {Promise<boolean>} `true` if at least one pending code exists,
   *   `false` otherwise or on database error.
   *
   * @example
   * const pending = await verificationService.hasPendingVerification('user@example.com');
   * if (pending) {
   *   return res.status(400).json({ code: 'PENDING_VERIFICATION' });
   * }
   */
  async hasPendingVerification(email) {
    try {
      const result = await query(
        `SELECT * FROM EmailVerificationCodes
         WHERE email = $1
         AND is_used = false
         AND expires_at > NOW()`,
        [email]
      );

      return result.rows.length > 0;
    } catch (error) {
      console.error('Error checking pending verification:', error);
      return false;
    }
  }
}

/**
 * Singleton instance of {@link VerificationService}.
 * Import this throughout the application instead of constructing a new instance.
 *
 * @type {VerificationService}
 */
export const verificationService = new VerificationService();
