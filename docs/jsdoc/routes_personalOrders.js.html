const router = express.Router();

const __filename = fileURLToPath(import.meta.url);
const __dirname  = path.dirname(__filename);
const UPLOADS_DIR = path.join(__dirname, '..', 'uploads');

// ── Multer — temporary storage for personal order file uploads ────────────────
const orderTempStorage = multer.diskStorage({
  destination: (_req, _file, cb) => {
    const dir = path.join(UPLOADS_DIR, 'temp');
    if (!fs.existsSync(dir)) { fs.mkdirSync(dir, { recursive: true }); }
    cb(null, dir);
  },
  filename: (_req, file, cb) => {
    const unique = `${Date.now()}-${Math.round(Math.random() * 1e9)}${path.extname(file.originalname)}`;
    cb(null, unique);
  },
});

const ORDER_ALLOWED_MIME_TYPES = new Set([
  'application/pdf',
  'application/msword',
  'application/vnd.openxmlformats-officedocument.wordprocessingml.document',
  'application/vnd.ms-powerpoint',
  'application/vnd.openxmlformats-officedocument.presentationml.presentation',
  'application/zip',
  'application/x-rar-compressed',
  'application/octet-stream',
  'image/jpeg',
  'image/png',
]);

const orderFileUpload = multer({
  storage: orderTempStorage,
  fileFilter: (_req, file, cb) => {
    if (ORDER_ALLOWED_MIME_TYPES.has(file.mimetype)) {
      cb(null, true);
    } else {
      cb(new Error(`File type not allowed: ${file.mimetype}`), false);
    }
  },
  limits: { fileSize: 50 * 1024 * 1024 }, // 50 MB
}).fields([{ name: 'files', maxCount: 10 }]);

// ── Helper: move uploaded file to the order directory ─────────────────────────
const moveToOrderDir = (file, orderId, baseUrl = 'https://localhost:5001') => {
  const destDir = path.join(UPLOADS_DIR, 'personal-orders', String(orderId));
  if (!fs.existsSync(destDir)) { fs.mkdirSync(destDir, { recursive: true }); }
  const destPath = path.join(destDir, file.filename);
  fs.renameSync(file.path, destPath);
  const relativePath = `/uploads/personal-orders/${orderId}/${file.filename}`;
  return baseUrl ? `${baseUrl}${relativePath}` : relativePath;
};

// ── Helper: clean up temp files on error ─────────────────────────────────────
const cleanupOrderTempFiles = (reqFiles) => {
  if (!reqFiles) { return; }
  for (const f of (reqFiles.files || [])) {
    if (fs.existsSync(f.path)) { fs.unlinkSync(f.path); }
  }
};

// ── Helper: resolve whether the authenticated user is admin ───────────────────
/**
 * Returns true if the given user has admin privileges.
 * @param {number} userId - Authenticated user ID from the JWT.
 * @returns {Promise<boolean>} True if the user is an admin, false otherwise.
 */
const isAdmin = async (userId) => {
  const result = await query(
    'SELECT is_admin FROM Users WHERE user_id = $1',
    [userId],
  );
  return result.rows[0]?.is_admin === true;
};

// ── GET /api/personal-orders ──────────────────────────────────────────────────
/**
 * Returns all personal orders belonging to the authenticated user.
 *
 * **Auth:** any authenticated user
 *
 * **Response:**
 * ```json
 * { "success": true, "personalOrders": [ { ...order } ] }
 * ```
 */
router.get('/', async (req, res) => {
  try {
    const result = await query(`
      SELECT
        po.*,
        pt.product_type_name  AS order_material_type,
        ac.age_category_name  AS order_material_age_category
      FROM PersonalOrders po
      LEFT JOIN ProductTypes   pt ON po.order_material_type_id         = pt.product_type_id
      LEFT JOIN AgeCategories  ac ON po.order_material_age_category_id = ac.age_category_id
      WHERE po.user_id = $1
      ORDER BY po.order_created_at DESC
    `, [req.userId]);

    res.json({ success: true, personalOrders: result.rows });
  } catch (error) {
    logger.error('Error fetching personal orders', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({
      success: false,
      error: { uk: 'Не вдалося отримати замовлення', en: 'Failed to fetch personal orders' },
    });
  }
});

// ── POST /api/personal-orders ─────────────────────────────────────────────────
/**
 * Creates a new personal order for the authenticated user.
 *
 * **Auth:** any authenticated user
 *
 * **Body:**
 * - `orderTitle`              {string}  Required
 * - `orderDescription`        {string}  Required
 * - `orderMaterialType`       {number}  Required — FK to ProductTypes.product_type_id
 * - `orderMaterialAgeCategory`{number}  Required — FK to AgeCategories.age_category_id
 * - `orderStatus`             {string}  Optional — defaults to `'Нове замовлення'`
 * - `orderPrice`              {number}  Optional — defaults to 0
 * - `orderDeadline`           {string}  Optional ISO 8601 timestamp
 *
 * **Response:** `201` with the created order row.
 */
router.post('/', async (req, res) => {
  const {
    orderTitle,
    orderDescription,
    orderStatus,
    orderMaterialType,
    orderMaterialAgeCategory,
    orderDeadline,
  } = req.body;

  if (!orderTitle || !orderDescription || !orderMaterialType || !orderMaterialAgeCategory) {
    return res.status(400).json({
      success: false,
      error: {
        uk: 'Відсутні обов\'язкові поля: orderTitle, orderDescription, orderMaterialType, orderMaterialAgeCategory',
        en: 'Missing required fields: orderTitle, orderDescription, orderMaterialType, orderMaterialAgeCategory',
      },
    });
  }

  try {
    const result = await query(`
      INSERT INTO PersonalOrders (
        user_id,
        order_title,
        order_description,
        order_status,
        order_price,
        order_material_type_id,
        order_material_age_category_id,
        order_deadline
      ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8)
      RETURNING *
    `, [
      req.userId,
      orderTitle,
      orderDescription,
      orderStatus || 'pending',
      null,
      orderMaterialType,
      orderMaterialAgeCategory,
      orderDeadline || null,
    ]);

    logger.info('Personal order created', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId: result.rows[0].order_id,
      userId: req.userId,
    });

    res.status(201).json({ success: true, personalOrder: result.rows[0] });
  } catch (error) {
    logger.error('Error creating personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({
      success: false,
      error: { uk: 'Не вдалося створити замовлення', en: 'Failed to create personal order' },
    });
  }
});

// ── POST /api/personal-orders/:orderId/files ──────────────────────────────────
/**
 * Uploads one or more files to an existing personal order (admin only).
 * Accepts `multipart/form-data` with field name `files` (up to 10 files).
 *
 * **Auth:** admin
 *
 * **Response:** `201` with the newly created file records.
 */
router.post('/:orderId/files', (req, res, next) => {
  orderFileUpload(req, res, (err) => {
    if (err instanceof multer.MulterError) {
      return res.status(400).json({ success: false, error: { uk: `Помилка завантаження: ${err.message}`, en: `Upload error: ${err.message}` } });
    }
    if (err) {
      return res.status(400).json({ success: false, error: { uk: err.message, en: err.message } });
    }
    next();
  });
}, async (req, res) => {
  const orderId = parseInt(req.params.orderId, 10);
  if (isNaN(orderId)) {
    cleanupOrderTempFiles(req.files);
    return res.status(400).json({ success: false, error: { uk: 'Некоректний ID замовлення', en: 'Invalid order ID' } });
  }

  const uploadedFiles = req.files?.files ?? [];
  if (uploadedFiles.length === 0) {
    return res.status(400).json({ success: false, error: { uk: 'Не надано жодного файлу', en: 'No files provided' } });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    if (!(await isAdmin(req.userId))) {
      await client.query('ROLLBACK');
      cleanupOrderTempFiles(req.files);
      return res.status(403).json({ success: false, error: { uk: 'Доступ лише для адміністраторів', en: 'Admins only' } });
    }

    const orderCheck = await client.query('SELECT order_id FROM PersonalOrders WHERE order_id = $1', [orderId]);
    if (orderCheck.rows.length === 0) {
      await client.query('ROLLBACK');
      cleanupOrderTempFiles(req.files);
      return res.status(404).json({ success: false, error: { uk: 'Замовлення не знайдено', en: 'Order not found' } });
    }

    const created = [];
    for (const file of uploadedFiles) {
      const fileUrl = moveToOrderDir(file, orderId);
      const fileResult = await client.query(`
        INSERT INTO Files (file_name, file_url, file_size)
        VALUES ($1, $2, $3)
        RETURNING file_id AS "fileId", file_name AS "fileName", file_url AS "fileUrl", file_size AS "fileSize"
      `, [file.originalname, fileUrl, file.size]);

      const inserted = fileResult.rows[0];
      await client.query(
        'INSERT INTO PersonalOrderFiles (order_id, file_id) VALUES ($1, $2)',
        [orderId, inserted.fileId],
      );
      created.push(inserted);
    }

    await client.query('COMMIT');

    logger.info('Files uploaded to personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      fileCount: created.length,
      userId: req.userId,
    });

    res.status(201).json({ success: true, files: created });
  } catch (error) {
    await client.query('ROLLBACK');
    cleanupOrderTempFiles(req.files);
    logger.error('Error uploading files to personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({ success: false, error: { uk: 'Не вдалося завантажити файли', en: 'Failed to upload files' } });
  } finally {
    client.release();
  }
});

// ── POST /api/personal-orders/:orderId/send-materials ────────────────────────
/**
 * Sends (or re-sends) personal order material download links to the order owner.
 *
 * Admin-only.  Fetches all files attached to the order and emails them to the
 * user who submitted the order.  Intended to be called after the admin has
 * finished uploading the completed order files.
 *
 * **Auth:** authenticated admin
 * @param {string} req.params.orderId - ID of the personal order.
 * @returns {object} 200 - `{ success: true, message: { uk, en } }`
 * @throws {ForbiddenError} 403 - Requester is not an admin.
 * @throws {NotFoundError}  404 - Order not found or has no files attached.
 */
router.post('/:orderId/send-materials', async (req, res, next) => {
  try {
    if (!(await isAdmin(req.userId))) {
      throw new ForbiddenError('Admin access required');
    }

    const orderId = parseInt(req.params.orderId, 10);
    if (isNaN(orderId) || orderId <= 0) {
      throw new NotFoundError('Order not found', { orderId: req.params.orderId });
    }

    const [orderDataResult, filesResult] = await Promise.all([
      query(
        `SELECT po.order_title, u.user_email
           FROM PersonalOrders po
           JOIN Users u ON u.user_id = po.user_id
          WHERE po.order_id = $1`,
        [orderId],
      ),
      query(
        `SELECT f.file_name AS "fileName", f.file_url AS "fileUrl"
           FROM Files f
           JOIN PersonalOrderFiles pof ON pof.file_id = f.file_id
          WHERE pof.order_id = $1
          ORDER BY f.file_id`,
        [orderId],
      ),
    ]);

    if (!orderDataResult.rows.length) {
      throw new NotFoundError('Order not found', { orderId });
    }

    if (!filesResult.rows.length) {
      throw new NotFoundError('No files found for this order', { orderId });
    }

    const { order_title, user_email } = orderDataResult.rows[0];

    await emailService.sendOrderMaterials(user_email, order_title, filesResult.rows);

    logger.info('Order materials sent by admin', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      adminUserId: req.userId,
    });

    res.json({
      success: true,
      message: {
        uk: 'Матеріали надіслано замовнику',
        en: 'Materials sent to the order owner',
      },
    });
  } catch (err) {
    next(err);
  }
});

// ── DELETE /api/personal-orders/:orderId/files/:fileId ────────────────────────
/**
 * Removes a file from a personal order and deletes it from disk (admin only).
 *
 * **Auth:** admin
 *
 * **Response:** `{ "success": true }`
 */
router.delete('/:orderId/files/:fileId', async (req, res) => {
  const orderId = parseInt(req.params.orderId, 10);
  const fileId  = parseInt(req.params.fileId, 10);

  if (isNaN(orderId) || isNaN(fileId)) {
    return res.status(400).json({ success: false, error: { uk: 'Некоректний ID', en: 'Invalid ID' } });
  }

  const client = await pool.connect();
  try {
    await client.query('BEGIN');

    if (!(await isAdmin(req.userId))) {
      await client.query('ROLLBACK');
      return res.status(403).json({ success: false, error: { uk: 'Доступ лише для адміністраторів', en: 'Admins only' } });
    }

    const fileRow = await client.query(
      'SELECT f.file_url FROM Files f JOIN PersonalOrderFiles pof ON pof.file_id = f.file_id WHERE pof.order_id = $1 AND f.file_id = $2',
      [orderId, fileId],
    );

    if (fileRow.rows.length === 0) {
      await client.query('ROLLBACK');
      return res.status(404).json({ success: false, error: { uk: 'Файл не знайдено', en: 'File not found' } });
    }

    await client.query('DELETE FROM PersonalOrderFiles WHERE order_id = $1 AND file_id = $2', [orderId, fileId]);
    await client.query('DELETE FROM Files WHERE file_id = $1', [fileId]);

    // Remove physical file
    const fileUrl = fileRow.rows[0].file_url;
    try {
      // fileUrl is an absolute URL like https://localhost:5001/uploads/...
      // Extract the path portion after the origin
      const urlPath = fileUrl.replace(/^https?:\/\/[^/]+/, '');
      const diskPath = path.join(__dirname, '..', urlPath);
      if (fs.existsSync(diskPath)) { fs.unlinkSync(diskPath); }
    } catch {
      // Non-fatal: DB is already cleaned up
    }

    await client.query('COMMIT');

    logger.info('File removed from personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      fileId,
      userId: req.userId,
    });

    res.json({ success: true, message: { uk: 'Файл успішно видалено', en: 'File deleted successfully' } });
  } catch (error) {
    await client.query('ROLLBACK');
    logger.error('Error deleting personal order file', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      fileId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({ success: false, error: { uk: 'Не вдалося видалити файл', en: 'Failed to delete file' } });
  } finally {
    client.release();
  }
});

// ── GET /api/personal-orders/:orderId ─────────────────────────────────────────
/**
 * Returns a single personal order by ID.
 * The authenticated user must own the order, or be an admin.
 *
 * **Auth:** owner or admin
 *
 * **Response:**
 * ```json
 * { "success": true, "personalOrder": { ...order } }
 * ```
 */
router.get('/:orderId', async (req, res) => {
  const { orderId } = req.params;

  try {
    const orderResult = await query(`
      SELECT
        po.*,
        pt.product_type_name  AS order_material_type,
        ac.age_category_name  AS order_material_age_category
      FROM PersonalOrders po
      LEFT JOIN ProductTypes   pt ON po.order_material_type_id         = pt.product_type_id
      LEFT JOIN AgeCategories  ac ON po.order_material_age_category_id = ac.age_category_id
      WHERE po.order_id = $1
    `, [orderId]);

    if (orderResult.rows.length === 0) {
      return res.status(404).json({
        success: false,
        error: { uk: 'Персональне замовлення не знайдено', en: 'Personal order not found' },
      });
    }

    const order = orderResult.rows[0];

    if (order.user_id !== req.userId) {
      if (!(await isAdmin(req.userId))) {
        return res.status(403).json({
          success: false,
          error: { uk: 'Немає доступу до цього замовлення', en: 'Not authorized to view this order' },
        });
      }
    }

    res.json({ success: true, personalOrder: order });
  } catch (error) {
    logger.error('Error fetching personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({
      success: false,
      error: { uk: 'Не вдалося отримати замовлення', en: 'Failed to fetch personal order' },
    });
  }
});

// ── PUT /api/personal-orders/:orderId ─────────────────────────────────────────
/**
 * Updates an existing personal order.
 * The authenticated user must own the order, or be an admin.
 * Only supplied fields are updated (PATCH semantics).
 *
 * **Auth:** owner or admin
 *
 * **Body (all optional):**
 * - `orderTitle`               {string}
 * - `orderDescription`         {string}
 * - `orderStatus`              {string}
 * - `orderDeclineReason`       {string} Required when `orderStatus` is `'declined'`; must be omitted otherwise.
 * - `orderPrice`               {number}
 * - `orderMaterialType`        {number} — FK to ProductTypes.product_type_id
 * - `orderMaterialAgeCategory` {number} — FK to AgeCategories.age_category_id
 * - `orderDeadline`            {string} — ISO 8601 timestamp
 *
 * **Response:** updated order row.
 */
router.put('/:orderId', async (req, res) => {
  const { orderId } = req.params;
  const {
    orderTitle,
    orderDescription,
    orderStatus,
    orderDeclineReason,
    orderPrice,
    orderMaterialType,
    orderMaterialAgeCategory,
    orderDeadline,
  } = req.body;

  try {
    const checkOwnership = await query(
      'SELECT user_id FROM PersonalOrders WHERE order_id = $1',
      [orderId],
    );

    if (checkOwnership.rows.length === 0) {
      return res.status(404).json({
        success: false,
        error: { uk: 'Персональне замовлення не знайдено', en: 'Personal order not found' },
      });
    }

    const orderUserId = checkOwnership.rows[0].user_id;

    if (orderUserId !== req.userId && !(await isAdmin(req.userId))) {
      return res.status(403).json({
        success: false,
        error: { uk: 'Немає доступу до цього замовлення', en: 'Not authorized to update this order' },
      });
    }

    if (orderStatus === 'declined' && !orderDeclineReason) {
      return res.status(400).json({
        success: false,
        error: {
          uk: 'При відхиленні замовлення необхідно вказати причину',
          en: 'A decline reason is required when setting status to declined',
        },
      });
    }

    const updates = [];
    const values  = [];
    let   idx     = 1;

    if (orderTitle               !== undefined) { updates.push(`order_title = $${idx++}`);                       values.push(orderTitle); }
    if (orderDescription         !== undefined) { updates.push(`order_description = $${idx++}`);                 values.push(orderDescription); }
    if (orderStatus              !== undefined) { updates.push(`order_status = $${idx++}`);                      values.push(orderStatus); }
    if (orderPrice               !== undefined) { updates.push(`order_price = $${idx++}`);                       values.push(orderPrice); }
    if (orderMaterialType        !== undefined) { updates.push(`order_material_type_id = $${idx++}`);            values.push(orderMaterialType); }
    if (orderMaterialAgeCategory !== undefined) { updates.push(`order_material_age_category_id = $${idx++}`);   values.push(orderMaterialAgeCategory); }
    if (orderDeadline            !== undefined) { updates.push(`order_deadline = $${idx++}`);                    values.push(orderDeadline); }

    // Keep order_decline_reason consistent with the DB check constraint:
    // declined → reason required (non-null); any other status → reason must be null.
    if (orderStatus !== undefined) {
      if (orderStatus === 'declined') {
        updates.push(`order_decline_reason = $${idx++}`);
        values.push(orderDeclineReason);
      } else {
        updates.push('order_decline_reason = NULL');
      }
    } else if (orderDeclineReason !== undefined) {
      updates.push(`order_decline_reason = $${idx++}`);
      values.push(orderDeclineReason);
    }

    if (updates.length === 0) {
      return res.status(400).json({
        success: false,
        error: { uk: 'Немає полів для оновлення', en: 'No fields to update' },
      });
    }

    values.push(orderId);
    const updateResult = await query(
      `UPDATE PersonalOrders SET ${updates.join(', ')} WHERE order_id = $${idx} RETURNING order_id`,
      values,
    );

    const result = await query(`
      SELECT
        po.*,
        pt.product_type_name  AS order_material_type,
        ac.age_category_name  AS order_material_age_category
      FROM PersonalOrders po
      LEFT JOIN ProductTypes   pt ON po.order_material_type_id         = pt.product_type_id
      LEFT JOIN AgeCategories  ac ON po.order_material_age_category_id = ac.age_category_id
      WHERE po.order_id = $1
    `, [updateResult.rows[0].order_id]);

    logger.info('Personal order updated', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      userId: req.userId,
    });

    res.json({ success: true, personalOrder: result.rows[0] });
  } catch (error) {
    logger.error('Error updating personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({
      success: false,
      error: { uk: 'Не вдалося оновити замовлення', en: 'Failed to update personal order' },
    });
  }
});

/**
 * Deletes a personal order.
 * The authenticated user must own the order, or be an admin.
 *
 * **Auth:** owner or admin
 *
 * **Response:** `{ "success": true, "message": { uk, en } }`
 */
router.delete('/:orderId', async (req, res) => {
  const { orderId } = req.params;

  try {
    const checkOwnership = await query(
      'SELECT user_id FROM PersonalOrders WHERE order_id = $1',
      [orderId],
    );

    if (checkOwnership.rows.length === 0) {
      return res.status(404).json({
        success: false,
        error: { uk: 'Персональне замовлення не знайдено', en: 'Personal order not found' },
      });
    }

    const orderUserId = checkOwnership.rows[0].user_id;

    if (orderUserId !== req.userId && !(await isAdmin(req.userId))) {
      return res.status(403).json({
        success: false,
        error: { uk: 'Немає доступу до цього замовлення', en: 'Not authorized to delete this order' },
      });
    }

    await query('DELETE FROM PersonalOrders WHERE order_id = $1', [orderId]);

    logger.info('Personal order deleted', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      userId: req.userId,
    });

    res.json({
      success: true,
      message: {
        uk: 'Замовлення успішно видалено',
        en: 'Personal order deleted successfully',
      },
    });
  } catch (error) {
    logger.error('Error deleting personal order', {
      module: 'routes/personalOrders',
      requestId: req.requestId,
      orderId,
      userId: req.userId,
      error: error.message,
    });
    res.status(500).json({
      success: false,
      error: { uk: 'Не вдалося видалити замовлення', en: 'Failed to delete personal order' },
    });
  }
});
